SYS_EXIT    equ 1
SYS_READ    equ 3
SYS_WRITE   equ 4
STDIN       equ 0
STDOUT      equ 1
STDOP       equ 2
CARAC       equ 10

section .bss
    nome resb CARAC

section .text
    global _start

_start:
    ; Pede ao usuário para digitar seu nome
    mov eax, SYS_WRITE
    mov ebx, STDOUT
    mov ecx, prompt
    mov edx, prompt_len
    int 0x80

    ; Lê o nome do usuário
    mov eax, SYS_READ
    mov ebx, STDIN
    mov ecx, nome
    mov edx, CARAC
    int 0x80

    ; Exibe a saudação
    mov eax, SYS_WRITE
    mov ebx, STDOUT
    mov ecx, greeting
    mov edx, greeting_len
    int 0x80

     ; Exibe o nome do usuário
    mov eax, SYS_WRITE
    mov ebx, STDOUT
    mov ecx, nome
    mov edx, CARAC
    int 0x80
    
    ; Sai do programa
    mov eax, STDOUT
    int 0x80

section .data
    ;cria mensagem para perguntar o nome
    prompt db "Digite seu nome: ", 0
    prompt_len equ $ - prompt
    
    ;cria mensagem de saudação
    greeting db "Saudações, ", 0
    greeting_len equ $ - greeting
